## 一、URL 处理


### 场景1：URL 构建

**传统写法：**

```
let urlString: String? = "https://example.com"

if let urlString = urlString,
   let url = URL(string: urlString){
    
    ///> 处理逻辑
}
```

**高阶函数处理：**

```
let url = urlString.flatMap(URL.init(string:))
if let url = url{
    
    ///> 处理逻辑
}
```


### 场景2：解析 URL Query 参数

**传统写法：**

``` swift
let url = URL(string: "https://example.com?name=alice&age=20&invalid")!

let queryItems = URLComponents(url: url, resolvingAgainstBaseURL: false)?
    .queryItems ?? []

var params:[String:String] = [:]
for item in queryItems {
    guard item.value != nil else {
        continue
    }
    params[item.name] = item.value
}
```

**高阶函数处理：**

```
let url = URL(string: "https://example.com?name=alice&age=20&invalid")!

let params = queryItems.reduce(into: [String:String]()) { partialResult, item in
    guard let value:String = item.value else{return}
    partialResult[item.name] = value
}
```


### 场景3：构建 URL 参数

**传统写法：**

```
let baseURL = "https://api.example.com/search"
var params = ["q": "swift", "page": "1"]
var queryItems:[URLQueryItem] = []
for param in params{
    queryItems.append(URLQueryItem(name: param.key, value: param.value))
}

var components = URLComponents(string: baseURL)
components?.queryItems = queryItems

let url = components?.url
```


**高阶函数处理：**

``` swift
let baseURL = "https://api.example.com/search"
let params = ["q": "swift", "page": "1"]

var components = URLComponents(string: baseURL)
components?.queryItems = params.map {
    URLQueryItem(name: $0.key, value: $0.value)
}

let url = components?.url
```

## 二、JSON 处理

### 场景1：过滤非法 JSON 数据

**传统写法：**

``` swift
let jsonArray: [[String: Any]] = [
    ["name": "Alice", "age": 20],
    ["name": "Bob"],
    ["age": 30]
]

struct User {
    let name: String
    let age: Int
}


var users:[User] = []
for dict in jsonArray{
    guard let name = dict["name"] as? String,
          let age = dict["age"] as? Int else {continue}
    
    users.append(User(name: name, age: age))
}

```

**高阶函数处理：**

```
let jsonArray: [[String: Any]] = [
    ["name": "Alice", "age": 20],
    ["name": "Bob"],
    ["age": 30]
]

struct User {
    let name: String
    let age: Int
}

let users = jsonArray.compactMap { dict -> User? in
    guard let name = dict["name"] as? String,
          let age = dict["age"] as? Int else {
        return nil
    }
    return User(name: name, age: age)
}
```

👉 关键点：\
- `compactMap` = 解析 + 过滤非法数据\
- 避免大量 `if let` 嵌套

------------------------------------------------------------------------



### 场景2：接口返回嵌套结构

``` swift
let response = [
    ["items": [1, 2, 3]],
    ["items": [4, 5]],
    ["items": [6, 7, 8]]
]

let allItems = response
    .compactMap { $0["items"] }
    .flatMap{$0}

print(allItems)\\[1, 2, 3, 4, 5, 6, 7, 8]
```

👉 `flatMap` 在这里的意义是：\
把「二维数组」压平成「一维数组」

## 三、数据模型转换（DTO → Domain → ViewModel）

### 场景：网络数据 → 业务模型 → UI 模型

``` swift
struct UserDTO {
    let id: Int
    let name: String
}

struct User {
    let id: Int
    let displayName: String
}

struct UserViewModel {
    let title: String
}

let dtos = [
    UserDTO(id: 1, name: "alice"),
    UserDTO(id: 2, name: "bob")
]

// DTO -> Domain
let users = dtos.map {
    User(id: $0.id, displayName: $0.name.capitalized)
}

// Domain -> ViewModel
let viewModels = users.map {
    UserViewModel(title: "User: \($0.displayName)")
}
```

👉 优势： - 每一层职责清晰 - 可测试性强 - 避免"上帝模型"

## 四、复杂数据流处理（链式调用）

### 场景：筛选 + 转换 + 排序

``` swift
struct Product {
    let name: String
    let price: Double
    let isAvailable: Bool
}

let products: [Product] = [
    Product(name: "Product1", price: 2, isAvailable: false),
    Product(name: "Product2", price: 1, isAvailable: true),
    Product(name: "Product3", price: 3, isAvailable: true)
]

let result = products
    .filter { $0.isAvailable }
    .sorted { $0.price < $1.price }
    .map { "\($0.name): \($0.price)" }

print(result)\\["Product2: 1.0", "Product3: 3.0"]
```

👉 一行完成数据流处理（声明式编程）

## 五、reduce 进阶：构建字典 / 分组

### 场景：数组转字典

``` swift
let users = [
    User(id: 1, displayName: "A"),
    User(id: 2, displayName: "B")
]

let dict = users.reduce(into: [Int: User]()) {
    $0[$1.id] = $1
}
```

### 场景：分组（Group By）

``` swift
struct Order {
    let userId: Int
    let price: Double
}

let orders: [Order] = [...]

let grouped = orders.reduce(into: [Int: [Order]]()) { result, order in
    result[order.userId, default: []].append(order)
}
```

## 结语

高阶函数的真正价值不在于"写得短"，而在于：

-   把**数据处理过程声明化**
-   把**业务逻辑从流程控制中解耦**
-   提升代码的**表达力和可维护性**

当你开始用 `map/filter/reduce` 去思考问题时，你已经在向函数式编程迈进了
